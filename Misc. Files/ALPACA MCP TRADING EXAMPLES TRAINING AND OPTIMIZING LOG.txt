# Function to calculate log trajectory metrics
def calculate_log_metrics(ticker_info):
    ticker, category = ticker_info
    try:
        # Download all available historical data
        data = yf.download(ticker, period='max', progress=False, show_errors=False)
        
        if data.empty or len(data) < 252:  # Require at least 1 year of data
            return None
        
        # Calculate log returns
        data['Log_Return'] = np.log(data['Close'] / data['Close'].shift(1))
        data['Cumulative_Log_Return'] = data['Log_Return'].cumsum()
        
        # Calculate metrics
        total_log_return = data['Cumulative_Log_Return'].iloc[-1]
        years = (data.index[-1] - data.index[0]).days / 365.25
        annualized_log_return = total_log_return / years
        
        # Volatility (annualized)
        volatility = data['Log_Return'].std() * np.sqrt(252)
        
        # Sharpe Ratio (assuming 0% risk-free rate for comparison)
        sharpe = annualized_log_return / volatility if volatility > 0 else 0
        
        # Recent performance (last 1, 3, 5 years)
        recent_1y = data['Cumulative_Log_Return'].iloc[-1] - data['Cumulative_Log_Return'].iloc[-252] if len(data) >= 252 else np.nan
        recent_3y = data['Cumulative_Log_Return'].iloc[-1] - data['Cumulative_Log_Return'].iloc[-756] if len(data) >= 756 else np.nan
        recent_5y = data['Cumulative_Log_Return'].iloc[-1] - data['Cumulative_Log_Return'].iloc[-1260] if len(data) >= 1260 else np.nan
        
        # Trend analysis: Linear regression on cumulative log returns
        x = np.arange(len(data))
        y = data['Cumulative_Log_Return'].values
        
        # Use GPU for linear regression if available
        if torch.cuda.is_available():
            x_tensor = torch.FloatTensor(x).unsqueeze(1).cuda()
            y_tensor = torch.FloatTensor(y).unsqueeze(1).cuda()
            
            # Add bias term
            X = torch.cat([x_tensor, torch.ones_like(x_tensor)], dim=1)
            
            # Solve using least squares: (X^T X)^-1 X^T y
            slope, intercept = torch.linalg.lstsq(X, y_tensor).solution[:2].cpu().numpy().flatten()
        else:
            slope, intercept = np.polyfit(x, y, 1)
        
        # Momentum score (recent trend strength)
        recent_slope = np.polyfit(np.arange(min(252, len(data))), 
                                   data['Cumulative_Log_Return'].iloc[-min(252, len(data)):].values, 1)[0]
        
        # Calculate max drawdown
        cumulative_returns = np.exp(data['Cumulative_Log_Return'])
        running_max = cumulative_returns.expanding().max()
        drawdown = (cumulative_returns - running_max) / running_max
        max_drawdown = drawdown.min()
        
        return {
            'Ticker': ticker,
            'Category': category,
            'Total_Log_Return': total_log_return,
            'Years': years,
            'Annualized_Log_Return': annualized_log_return,
            'Volatility': volatility,
            'Sharpe_Ratio': sharpe,
            'Recent_1Y': recent_1y,
            'Recent_3Y': recent_3y,
            'Recent_5Y': recent_5y,
            'Long_Term_Slope': slope,
            'Recent_Momentum': recent_slope,
            'Max_Drawdown': max_drawdown,
            'Data_Points': len(data),
            'Start_Date': data.index[0].strftime('%Y-%m-%d'),
            'End_Date': data.index[-1].strftime('%Y-%m-%d')
        }
    
    except Exception as e:
        return None

def calculate_log_metrics_robust(ticker_info):
    """Robust calculation with retry logic and better error handling"""
    ticker, category = ticker_info
    
    for attempt in range(3):  # Try up to 3 times
        try:
            stock = yf.Ticker(ticker)
            data = stock.history(period='max', timeout=10)
            
            if data is None or len(data) == 0:
                if attempt < 2:
                    time.sleep(0.5)  # Brief pause before retry
                    continue
                return None
            
            if len(data) < 252:
                return None
            
            # Clean data
            data = data[['Close']].copy()
            data = data.dropna()
            
            if len(data) < 252:
                return None
            
            # Calculate log returns
            data['Log_Return'] = np.log(data['Close'] / data['Close'].shift(1))
            data = data.dropna()
            data['Cumulative_Log_Return'] = data['Log_Return'].cumsum()
            
            # Core metrics
            total_log_return = float(data['Cumulative_Log_Return'].iloc[-1])
            years = (data.index[-1] - data.index[0]).days / 365.25
            annualized_log_return = total_log_return / years if years > 0 else 0
            volatility = float(data['Log_Return'].std() * np.sqrt(252))
            sharpe = annualized_log_return / volatility if volatility > 0 else 0
            
            # Recent performance
            n = len(data)
            recent_1y = float(data['Cumulative_Log_Return'].iloc[-1] - 
                            data['Cumulative_Log_Return'].iloc[-min(252, n)]) if n >= 252 else np.nan
            recent_3y = float(data['Cumulative_Log_Return'].iloc[-1] - 
                            data['Cumulative_Log_Return'].iloc[-min(756, n)]) if n >= 756 else np.nan
            
            # Trend analysis with GPU
            x = np.arange(len(data)).astype(np.float32)
            y = data['Cumulative_Log_Return'].values.astype(np.float32)
            
            if torch.cuda.is_available() and len(data) > 500:
                try:
                    x_gpu = torch.from_numpy(x).cuda().unsqueeze(1)
                    y_gpu = torch.from_numpy(y).cuda().unsqueeze(1)
                    X = torch.cat([x_gpu, torch.ones_like(x_gpu)], dim=1)
                    slope = float(torch.linalg.lstsq(X, y_gpu).solution[0].cpu().numpy())
                except:
                    slope = float(np.polyfit(x, y, 1)[0])
            else:
                slope = float(np.polyfit(x, y, 1)[0])
            
            # Recent momentum
            recent_n = min(252, len(data))
            recent_slope = float(np.polyfit(np.arange(recent_n), 
                                          data['Cumulative_Log_Return'].iloc[-recent_n:].values, 1)[0])
            
            # Max drawdown
            cum_ret = np.exp(data['Cumulative_Log_Return'])
            running_max = cum_ret.expanding().max()
            drawdown = (cum_ret - running_max) / running_max
            max_dd = float(drawdown.min())
            
            return {
                'Ticker': ticker,
                'Category': category,
                'Total_Log_Return': total_log_return,
                'Years': round(years, 1),
                'Annualized_Log_Return': annualized_log_return,
                'Volatility': volatility,
                'Sharpe_Ratio': sharpe,
                'Recent_1Y': recent_1y,
                'Recent_3Y': recent_3y,
                'Long_Term_Slope': slope,
                'Recent_Momentum': recent_slope,
                'Max_Drawdown': max_dd,
                'Data_Points': len(data),
                'Start_Date': data.index[0].strftime('%Y-%m-%d'),
                'End_Date': data.index[-1].strftime('%Y-%m-%d')
            }
            
        except Exception as e:
            if attempt < 2:
                time.sleep(0.5)
                continue
            return None
    
    return None
